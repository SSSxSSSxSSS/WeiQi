func test_capture_single_stone():
    var board = Board.new()
    board.set_stone(2, 3, Stone.Type.WHITE)  # 被提目标
    board.set_stone(2, 2, Stone.Type.BLACK)  # 包围
    board.set_stone(2, 4, Stone.Type.BLACK)
    board.set_stone(3, 3, Stone.Type.BLACK)
    # (2,3) 只剩一口气在 (1,3)
    var rules = GoRules.new()
    var result = rules.play_move(board, 1, 3, Stone.Type.BLACK)
    assert_true(result.valid)
    assert_eq(result.captured.size(), 1)
    assert_eq(board.get_stone(2, 3), Stone.Type.EMPTY, "白子应被移除")

func test_capture_multiple_groups():
    var board = Board.new()
    board.set_stone(2, 3, Stone.Type.WHITE)
    board.set_stone(2, 2, Stone.Type.BLACK)
    board.set_stone(2, 4, Stone.Type.BLACK)
    board.set_stone(3, 3, Stone.Type.BLACK)
    board.set_stone(4, 3, Stone.Type.WHITE)
    board.set_stone(4, 2, Stone.Type.BLACK)
    board.set_stone(4, 4, Stone.Type.BLACK)
    board.set_stone(3, 3, Stone.Type.BLACK)
    # 两个白子各自只剩一口气
    var rules = GoRules.new()
    var result = rules.play_move(board, 1, 3, Stone.Type.BLACK)
    assert_true(result.valid)
    assert_eq(result.captured.size(), 2)

- [ ] **步骤 3: 实现提子逻辑**

```gdscript
# 在 play_move 落子后追加：
# 对 4 个邻居遍历，对方棋组无气则移除
for nb in board.get_neighbors(row, col):
    var nb_stone := board.get_stone(nb.x, nb.y)
    if nb_stone != Stone.Type.EMPTY and nb_stone != color:
        var group := StoneGroup.build_group(board, nb)
        if group["liberties"].size() == 0:
            for s in group["stones"]:
                board.set_stone(s.x, s.y, Stone.Type.EMPTY)
                captured.append(s)
```
