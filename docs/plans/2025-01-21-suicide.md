func test_suicide_is_invalid():
    var board = Board.new()
    board.set_stone(2, 3, Stone.Type.WHITE)
    board.set_stone(4, 3, Stone.Type.WHITE)
    board.set_stone(3, 2, Stone.Type.WHITE)
    board.set_stone(3, 4, Stone.Type.WHITE)
    var rules = GoRules.new()
    var result = rules.play_move(board, 3, 3, Stone.Type.BLACK)
    assert_false(result.valid)
    assert_eq(result.reason, "suicide")

func test_not_suicide_if_can_capture():
    var board = Board.new()
    board.set_stone(2, 3, Stone.Type.WHITE)
    board.set_stone(1, 2, Stone.Type.BLACK)
    board.set_stone(1, 4, Stone.Type.BLACK)
    board.set_stone(3, 2, Stone.Type.BLACK)
    board.set_stone(3, 4, Stone.Type.BLACK)
    var rules = GoRules.new()
    var result = rules.play_move(board, 2, 2, Stone.Type.BLACK)
    assert_true(result.valid)
